## Clock (100 MHz)
set_property PACKAGE_PIN Y9 [get_ports clk]
set_property IOSTANDARD LVCMOS33 [get_ports clk]
create_clock -period 10.000 -name sys_clk [get_ports clk]

## IR Sensor Input (example: JA1)
set_property PACKAGE_PIN Y11 [get_ports ir_in]
set_property IOSTANDARD LVCMOS33 [get_ports ir_in]

## UART TX (example: JA2)
set_property PACKAGE_PIN AA11 [get_ports tx]
set_property IOSTANDARD LVCMOS33 [get_ports tx]

## Debug LED (example: LD0)
set_property PACKAGE_PIN T22 [get_ports debug_led]
set_property IOSTANDARD LVCMOS33 [get_ports debug_led]
